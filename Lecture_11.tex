\subsection{Reduction of Order}
\slide[Reduction of order]{
Given\[p(t)y\pp + q(t)y\p+ r(t)y=0\] and one solution $y_1(t)$, we want to find the second solution $y_2(t)$.\vfill
\enum{\item Guess $y_2(t) = a(t)y_1(t)$
\item Substitute into ODE, rearrange to get an ODE for $a\p(t)=v(t)$
\item Solve for $v(t)$ using integrating factors
\item Integrate $v(t)$ to find $a(t)$ and $y_2(t)$}\vfill
\[y_h=c_1y_1(t)+c_2y_2(t)\]
}

\slide[Inhomogeneous ODEs]{
Given\[p(t)y\pp + q(t)y\p+ r(t)y=h(t)\] and one solution $y_1(t)$, we want to find the general solution.
\vfill
\enum{\item Find $y_2(t)$ by reduction of order
\item Use the method of undetermined coefficients to find the particular part of the solution
\subitem{Watch out for mathematical resonance}
}\vfill
\[y_g=y_p+c_1y_1(t)+c_2y_2(t)\]
}
