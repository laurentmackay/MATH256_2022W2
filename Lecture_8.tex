\section{Lecture 8}
\subsection{Introduction}
\slide[Review: Method of Undetermined Coefficients]{
$ay\pp+by\p+cy=h(t)\neq0 \qquad  \Rightarrow \quad  y = y_p(t) + c_1y_1(t)+c_2y_2(t) $\vfill
How to find $y_p$:
\enum{
\item Differentiate $h(t)$ to get functional forms \student{\[\{f_1(t), f_2(t), f_3(t),\dots \}\]}
\item If there are a finite number of functions, guess \[y_p = \student{Af_1(t) +Bf_2(t) + Cf_3(t)+\dots}\]
\item Plug $y_p$ into ODE
\item Solve for undetermined coefficients ($A, B, C\dots$)
}
}

\slide[Solving for undetermined coefficients ($A, B, C\dots$) ]{
$ay\pp+by\p+cy=h(t)\neq0 \qquad  \Rightarrow \quad  y = y_p + c_1y_1(t)+c_2y_2(t) $
 \[y_p = Af_1(t) +Bf_2(t) + Cf_3(t)+\dots\]
We obtain a linear system of equations, $n$ equations $n$ unknowns 
\[\mathbf{M} \mat{c}{A\\B\\C\\\vdots} = \mathbf{v}.\]\vfill
\student{If any of the functions $f_i$ are linearly dependent with $y_1(t)$ or $y_2(t)$, we have an underdetermined system of equations...we won't find a unique solution. 
\vfill
This situation is called mathematical resonance.
}
}
\subsection{Mathematical Resonance}
\slide[Dealing with mathematical resonance]{\vspace{-.5em}
$ay\pp+by\p+cy=h(t)\neq0 \qquad  \Rightarrow \quad  y = y_p + c_1y_1(t)+c_2y_2(t) $\vfill
Suppose the family of functional for $h(t)$ is \[\{f_1(t), f_2(t), f_3(t) \}\]  where $f_3(t)$  and $y_1(t)$ are linearly dependent. \vfill 
\student{Recall: when we had repeated roots for the homogeneous case, we obtained a new linearly dependent function by multiplying by $t$.}\vfill
Then we should guess
\[y_p = \student{ Af_1(t) + Bf_2(t) +Ctf_3(t)}\]
\student{If $tf_3(t)$ is linearly dependent with $y_2(t)$, then use  $t^2f_3(t)$ instead.}
}

\slide[A simple example of mathematical resonance]{\vspace{-2em}
\algn{y\pp + y &= h(t)=\sin \omega t \\ y_h &= c_1\cos( t) + c_2\sin (t)}
\student{Family of functional forms for $h(t)$\[\{\cos(\omega t), \sin(\omega t)\}\]
Naive guess:\[y_p =  A\cos(\omega t)+B\sin(\omega t) \]\vfill
Notice that if $\omega=1$, then $y_p=y_h$ \hfill (not good!)\vfill
\twomini[.2]{.5}{.5}{\algn{&\uline{\omega  \neq 1}\\
y_p &= A\cos(\omega t)+B\sin(\omega t) }}{\algn{&\uline{\omega  = 1}\\
y_p &= At\cos(\omega t)+Bt\sin(\omega t) }}
}
}

\slide{Find the general solution of $y\pp+5y\p+4y=e^{-4t}$
\student{\algn{r_{1,2}&=\frac{-5\pm\sqrt{25-16}}{2} = \frac{-5\pm3}{2} = -1, -4\\
y_h &= c_1 e^{-t} +\underbrace{c_2e^{-4t}}_{\propto h(t)}\\
\text{Try: }\quad y_p&=Ate^{-4t}\\
y_p\p &= A \left( e^{-4t} - 4te^{-4t}\right)\\
y_p\pp &= -Ae^{-4t} -4A \left(e^{-4t}  - 4te^{-4t}\right)\\
&= -8Ae^{-4t} + 16 Ate^{-4t}
}
}
}
\slide{
\student{\algn{\intertext{plug into DE:}
-8Ae^{-4t} +16 Ate^{-4t} &+ 5Ae^{-4t} -20Ate^{-4t} + 4Ate^{-4t} =e^{-4t}\\
(-8+5)Ae^{-4t} +& (20-20)te^{-4t} =e^{-4t}\\
-3Ae^{-4t} &=e^{-4t}\\\\
A&=-\frac13\\\\
y &=c_1e^{-4t}+c_2e^{-t} -\frac13te^{-4t}
}}}


\slide{Find the general solution of $y\pp+4y\p+4y=e^{-2t}$
\student{\algn{r_{1,2}&=\frac{-4\pm\sqrt{16-16}}{2} = -2\\

&=4 At^2 e^{-2 t} -8 At e^{-2 t} +2 A e^{-2 t}
}
}}
\slide{\student{\algn{ \intertext{plug into DE:}
